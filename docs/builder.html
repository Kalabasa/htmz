htmz builder

<script>
  function htmz(frame) {
    // ------------------------------------------------------------------------
    // Multiple targets (a.k.a out of band updates)
    // ------------------------------------------------------------------------
    // Following the HTML spec which states that element `id`s must be unique,
    // this extension replaces existing elements with matching elements
    // in the response if they have the same `id`
    for (newElement of frame.contentDocument.querySelectorAll("[id]")) {
      document.getElementById(newElement.id)?.replaceWith(newElement);
    }
    // ------------------------------------------------------------------------

    // Remove setTimeout to let the browser autoscroll content changes into view
    setTimeout(() =>
      document
        .querySelector(frame.contentWindow.location.hash || null)
        ?.replaceWith(...frame.contentDocument.body.children)
    );
  }
</script>
<iframe hidden name=htmz onload="htmz(this)"></iframe>
